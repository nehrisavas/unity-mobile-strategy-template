# VERITABANI SEMASI

```sql
-- Ulus Tanim Tablosu
CREATE TABLE Nations (
    NationId TINYINT PRIMARY KEY,
    NationName NVARCHAR(50) NOT NULL,
    NationCode VARCHAR(10) NOT NULL,        -- 'OTTOMAN', 'ENGLAND', etc.
    ColorPrimary VARCHAR(7) NOT NULL,        -- Hex color
    ColorSecondary VARCHAR(7) NOT NULL,
    IconPath VARCHAR(255),
    Description NVARCHAR(500)
);

-- Ulus Bonuslari
CREATE TABLE NationBonuses (
    BonusId INT PRIMARY KEY IDENTITY,
    NationId TINYINT NOT NULL,
    BonusType VARCHAR(50) NOT NULL,          -- 'INFANTRY_ATTACK', 'GOLD_PRODUCTION', etc.
    BonusValue DECIMAL(5,2) NOT NULL,        -- 0.15 = %15
    Description NVARCHAR(200),

    FOREIGN KEY (NationId) REFERENCES Nations(NationId)
);

-- Ozel Birimler
CREATE TABLE NationSpecialUnits (
    UnitId INT PRIMARY KEY IDENTITY,
    NationId TINYINT NOT NULL,
    UnitName NVARCHAR(50) NOT NULL,
    BaseUnitType VARCHAR(20) NOT NULL,       -- 'INFANTRY', 'CAVALRY', 'ARCHER'
    AttackBonus DECIMAL(5,2),
    DefenseBonus DECIMAL(5,2),
    SpeedModifier DECIMAL(5,2),
    SpecialAbility NVARCHAR(200),
    CostMultiplier DECIMAL(3,1) DEFAULT 2.0,

    FOREIGN KEY (NationId) REFERENCES Nations(NationId)
);

-- Ozel Binalar
CREATE TABLE NationSpecialBuildings (
    BuildingId INT PRIMARY KEY IDENTITY,
    NationId TINYINT NOT NULL,
    BuildingName NVARCHAR(50) NOT NULL,
    Effect NVARCHAR(200),
    BonusType VARCHAR(50),
    BonusValue DECIMAL(5,2),
    WoodCost INT,
    StoneCost INT,
    IronCost INT,
    GoldCost INT,

    FOREIGN KEY (NationId) REFERENCES Nations(NationId)
);

-- Oyuncu Ulus Secimi
ALTER TABLE Players ADD NationId TINYINT FOREIGN KEY REFERENCES Nations(NationId);
```

# DENGE NOTLARI

## Tasarim Ilkeleri
1. **Hicbir ulus mutlak guclu olmamali** - Her ulusun zayifligi var
2. **Counter sistemi** - Her guc bir baska ulusa karsi zayif
3. **Oyun stili farki** - Agresif vs savunmaci vs dengeli
4. **Esit ekonomi** - Kaynak bonuslari dengeli
